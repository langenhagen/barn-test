284  const unsigned int dots_to_add_int = static_cast<unsigned int>(dots_to_add_float);

286  dots_to_add_float -= dots_to_add_int;

289  DurationType dur;

292  const auto result = call(fun_, arg_tuple, dur);
293 
294  if (comp_(result, reference_result)) {
