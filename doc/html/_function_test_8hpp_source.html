189  TestReturnType test(
190  const std::string& test_name,
191  const ResultType& expected_result,
192  const ArgTypes&... args) {
193 
194  TestReturnType ret;
195  get_passed(ret) = false;
196 
197  std::string output = "TESTING " + test_name + ": ";
198  output.resize(output_line_length, '.');
199  os_ << output << " ";
200 
201  try {
202  using namespace std::chrono;
203  const auto clock_start = steady_clock::now();
204  const ResultType result = fun_(args...);
205  get_result(ret) = result;
206  const auto ms = duration_cast< milliseconds>(steady_clock::now() - clock_start);
207 
208  if (comp_(result, expected_result)) {
209  ++n_passed_tests_;
210  is_last_test_passed_ = true;
211 
212  os_ << "OK (" << ms.count() << " ms)\n";
213 
214  get_passed(ret) = true;
215  }
216  else {
217  is_last_test_passed_ = false;
218  is_all_tests_passed_ = false;
219 
220  os_ << "FAILED (" << ms.count() << " ms)\n";
221 
222  if (verbose) {
223  os_ << " RESULT: " << to_string_function_(result) << "\n"
224  << " EXPECTED: " << to_string_function_(expected_result) << "\n"
225  << ".\n";
226  }
227  }
228 
229  ++n_tests_;
230  last_test_result_ = result;
231  last_invocation_duration_ = ms;
232  get_duration(ret) = ms;
233  accumulated_invocation_durations_ += ms;
234 
235  }
236  catch (std::exception& ex) {
237  os_ << "EXCEPTION\n"
238  << typeid(ex).name() << ":\n" << ex.what() << std::endl;
239  }
240  catch (...) {
241  os_ << "EXCEPTION\n"
242  << "unknown\n";
243  }
244 
245  return ret;
246  }
